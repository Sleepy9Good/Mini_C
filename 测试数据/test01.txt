/* test float, multiple level token table*/

float x[10];

void main(void)
{
	int num;
	num = 0;
	x[0] = 1.0;
	do{
		int t;
		{
			float n;
			num  = num +1;
			x[num] = x[num -1] + 0.5;
			n = x[num];
			t = n;
			output(t);
		}
	}while(num<8);
	num = 0;
	do{
		float t[10];
		{
			int n;
			num  = num +1;
			x[num] = x[num -1] + 1;
			t[9] = x[num];
			n = t[9];
			output(n);
		}
	}while(num<8);
}
