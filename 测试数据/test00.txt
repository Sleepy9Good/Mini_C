int x[10];

int Findmin(int a[], int low, int high)
{
	int i; int x; int k;
	k = low;
	x = a[low];
	i = low + 1;
	do
	{
		if (a[i] < x)
		{
			x = a[i];
			k = i;
		}
		i = i + 1;
	} while (i < high);
	return k;
}

void Selectsort(int a[], int low, int high)
{
	int i; int k;
	i = low;
	do
	{
		int t;
		k = Findmin(a, i, high);
		t = a[k];
		a[k] = a[i];
		a[i] = t;
		i = i + 1;
	} while (i < high - 1);
}

void main(void)
{
	int i;
	i = 0;
	do
	{
		x[i] = input();
		i = i + 1;
	} while (i < 10);
	Selectsort(x, 0, 10);
	i = 0;
	do
	{
		output(x[i]);
		i = i + 1;
	} while (i < 10);
}
